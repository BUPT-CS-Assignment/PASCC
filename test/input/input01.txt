program TestArithmetic(void);
var
  a, b: integer;
  c, d: real;
begin
  a := 10;
  b := 3;
  c := 10.5;
  d := 2.5;
  writeln(a + b); 
  writeln(a - b); 
  writeln(a * b); 
  writeln(a / b); 
  writeln(a div b); 
  writeln(a mod b); 
  writeln(c + d); 
  writeln(c - d); 
  writeln(c * d); 
  writeln(c / d); 
end.
